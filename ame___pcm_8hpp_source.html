285  void process (AudioBlockView<FloatType, N>& block)
286  {
287  if (playing.load() == false)
288  {
289  return;
290  }
291 
292  assert (block.getNumChannels() >= numChannels);
293  assert (0 <= readPosition.load() && readPosition <= numSamples);
294 
296  switch (formatTag)
297  {
298  case fmt::wFormatTag::PCM:
299  {
300  auto offset = (bitRate / 8) * numChannels * readPosition.load();
301  const auto nSamp = block.getNumSamplesPerChannel();
302  for (uint_fast32_t samp = 0; samp < nSamp; ++samp)
303  {
304  if (readPosition + samp >= numSamples)
305  {
307  readPosition.store (0);
308  playing.store (false);
309  return;
310  }
311 
312  for (uint_fast32_t ch = 0; ch < block.getNumChannels(); ++ch)
313  {
314  int32_t temp = 0;
315  for (uint32_t b = 32 - bitRate; b < 32; b += 8) //bitDepthが16bitや24bitのときは下部を0埋めする.
316  {
317  temp |= (dataChunk.data[offset] << b);
318  ++offset;
319  }

321  block.addSample (ch, samp, s);
322  }
323  }

326  }
