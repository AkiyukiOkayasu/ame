173  if (pa != pb)
174  {
175  throw (Error(EQUAL_ERROR,nb));

180 template <typename T>
181 void assert_equal(unsigned long nb,AnyPattern<T> &pa, AnyPattern<T> &pb)
182 {
183  ASSERT_NOT_EMPTY(pa);
184  ASSERT_NOT_EMPTY(pb);
185 
186  if (pa.nbSamples() != pb.nbSamples())
187  {
188  throw (Error(EQUAL_ERROR,nb));
189  }
190 
191  unsigned long i=0;
192  char id[40];
193 
194  T *ptrA = pa.ptr();
195  T *ptrB = pb.ptr();
196 
197  for(i=0; i < pa.nbSamples(); i++)
198  {
199  try
200  {
201  assert_equal(nb,ptrA[i],ptrB[i]);
202  }
203  catch(Error &err)
204  {
205  sprintf(id," (nb=%lu)",i+1);
206  strcat(err.details,id);
207  throw(err);
208  }
209  }
210 };
