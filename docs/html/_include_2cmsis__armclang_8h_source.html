914 __STATIC_FORCEINLINE uint32_t __ROR(uint32_t op1, uint32_t op2)
915 {
916  op2 %= 32U;
917  if (op2 == 0U)
918  {
919  return op1;
920  }
921  return (op1 >> op2) | (op1 << (32U - op2));
922 }
