468 __STATIC_INLINE __ASM void __FPU_Enable(void)
469 {
470  ARM
471 
472  //Permit access to VFP/NEON, registers by modifying CPACR
473  MRC p15,0,R1,c1,c0,2
474  ORR R1,R1,#0x00F00000
475  MCR p15,0,R1,c1,c0,2
476 
477  //Ensure that subsequent instructions occur in the context of VFP/NEON access permitted
478  ISB
479 
480  //Enable VFP/NEON
481  VMRS R1,FPEXC
482  ORR R1,R1,#0x40000000
483  VMSR FPEXC,R1
484 
485  //Initialise VFP/NEON registers to 0
486  MOV R2,#0
487 
488  //Initialise D16 registers to 0
489  VMOV D0, R2,R2
490  VMOV D1, R2,R2
491  VMOV D2, R2,R2
492  VMOV D3, R2,R2
493  VMOV D4, R2,R2
494  VMOV D5, R2,R2
495  VMOV D6, R2,R2
496  VMOV D7, R2,R2
497  VMOV D8, R2,R2
498  VMOV D9, R2,R2
499  VMOV D10,R2,R2
500  VMOV D11,R2,R2
501  VMOV D12,R2,R2
502  VMOV D13,R2,R2
503  VMOV D14,R2,R2
504  VMOV D15,R2,R2
505 
506  IF {TARGET_FEATURE_EXTENSION_REGISTER_COUNT} == 32
507  //Initialise D32 registers to 0
508  VMOV D16,R2,R2
509  VMOV D17,R2,R2
510  VMOV D18,R2,R2
511  VMOV D19,R2,R2
512  VMOV D20,R2,R2
513  VMOV D21,R2,R2
514  VMOV D22,R2,R2

517  VMOV D25,R2,R2
518  VMOV D26,R2,R2
519  VMOV D27,R2,R2
520  VMOV D28,R2,R2
521  VMOV D29,R2,R2
522  VMOV D30,R2,R2
523  VMOV D31,R2,R2
524  ENDIF

528  LDR R2,=0x00086060 //Mask off all bits that do not have to be preserved. Non-preserved bits can/should be zero.
