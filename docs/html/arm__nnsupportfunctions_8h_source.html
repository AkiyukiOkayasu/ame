678 __STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult(const q31_t m1, const q31_t m2)
679 {
680  q31_t result = 0;
681  // Rounding offset to add for a right shift of 31
682  q63_t mult = 1 << 30;
683 
684  if ((m1 < 0) ^ (m2 < 0))
685  {
686  mult = 1 - mult;
687  }
688  // Gets resolved as a SMLAL instruction
689  mult = mult + (q63_t)m1 * m2;
690 
691  // Utilize all of the upper 32 bits. This is the doubling step
692  // as well.
693  result = (int32_t) (mult / (1ll << 31));
694 
695  if ((m1 == m2) && (m1 == (int32_t)Q31_MIN))
696  {
697  result = Q31_MAX;
698  }
699  return result;
700 }

716 __STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult_no_sat(const q31_t m1, const q31_t m2)
717 {
718  q31_t result = 0;
719  union arm_nn_long_long mult;
720 
721  // Rounding offset to add for a right shift of 31
722  mult.word.low = 1 << 30;
723  mult.word.high = 0;
724 
725  // Gets resolved as a SMLAL instruction
726  mult.long_long = mult.long_long + (q63_t)m1 * m2;
727

729  // as well.
730  result = (int32_t)(mult.long_long >> 31);

732  return result;

743 __STATIC_FORCEINLINE q31_t arm_nn_divide_by_power_of_two(const q31_t dividend, const q31_t exponent)
744 {
745  q31_t result = 0;
746  const q31_t remainder_mask = (1 << exponent) - 1;
747  int32_t remainder = remainder_mask & dividend;
748 
749  // Basic division
750  result = dividend >> exponent;
751 
752  // Adjust 'result' for rounding (mid point away from zero)
753  q31_t threshold = remainder_mask >> 1;
754  if (result < 0)
755  {
756  threshold++;
757  }
758  if (remainder > threshold)
759  {
760  result++;
761  }
762 
763  return result;
764 }
