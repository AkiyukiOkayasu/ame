196 __inline static U32 rt_inc_qi (U32 size, U8 *count, U8 *first) {
197  U32 cnt,c2;
198 #ifdef __USE_EXCLUSIVE_ACCESS
199  do {
200  if ((cnt = __ldrex(count)) == size) {
201  __clrex();
202  return (cnt); }
203  } while (__strex(cnt+1U, count));
204  do {
205  c2 = (cnt = __ldrex(first)) + 1U;
206  if (c2 == size) { c2 = 0U; }
207  } while (__strex(c2, first));
208 #else
209  __disable_irq();
210  if ((cnt = *count) < size) {
211  *count = (U8)(cnt+1U);
212  c2 = (cnt = *first) + 1U;
213  if (c2 == size) { c2 = 0U; }
214  *first = (U8)c2;
215  }
216  __enable_irq ();
217 #endif
218  return (cnt);
219 }
