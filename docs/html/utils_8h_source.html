50  __STATIC_FORCEINLINE uint32_t arm_recip_q31(
51  q31_t in,
52  q31_t * dst,
53  const q31_t * pRecipTable)
54  {
55  q31_t out;
56  uint32_t tempVal;
57  uint32_t index, i;
58  uint32_t signBits;
59 
60  if (in > 0)
61  {
62  signBits = ((uint32_t) (__CLZ( in) - 1));
63  }
64  else
65  {
66  signBits = ((uint32_t) (__CLZ(-in) - 1));
67  }
68 
69  /* Convert input sample to 1.31 format */
70  in = (in << signBits);
71 
72  /* calculation of index for initial approximated Val */
73  index = (uint32_t)(in >> 24);
74  index = (index & INDEX_MASK);
75 
76  /* 1.31 with exp 1 */
77  out = pRecipTable[index];
78 
79  /* calculation of reciprocal value */
80  /* running approximation for two iterations */
81  for (i = 0U; i < 2U; i++)
82  {
83  tempVal = (uint32_t) (((q63_t) in * out) >> 31);
84  tempVal = 0x7FFFFFFFu - tempVal;
85  /* 1.31 with exp 1 */
86  /* out = (q31_t) (((q63_t) out * tempVal) >> 30); */
87  out = clip_q63_to_q31(((q63_t) out * tempVal) >> 30);
88  }
89 
90  /* write output */
91  *dst = out;
92 
93  /* return num of signbits of out = 1/in value */
94  return (signBits + 1U);
95  }

155 __STATIC_INLINE void arm_norm_64_to_32u(uint64_t in, int32_t * normalized, int32_t *norm)
156 {
157  int32_t n1;
158  int32_t hi = (int32_t) (in >> 32);
159  int32_t lo = (int32_t) ((in << 32) >> 32);
160 
161  n1 = __CLZ(hi) - 32;
162  if (!n1)
163  {
164  /*
165  * input fits in 32-bit
166  */
167  n1 = __CLZ(lo);
168  if (!n1)
169  {
170  /*
171  * MSB set, need to scale down by 1
172  */
173  *norm = -1;
174  *normalized = (((uint32_t) lo) >> 1);
175  } else
176  {
177  if (n1 == 32)
178  {
179  /*
180  * input is zero
181  */
182  *norm = 0;
183  *normalized = 0;
184  } else
185  {
186  /*
187  * 32-bit normalization
188  */
189  *norm = n1 - 1;
190  *normalized = lo << *norm;
191  }
192  }
193  } else
194  {
195  /*
196  * input fits in 64-bit
197  */
198  n1 = 1 - n1;
199  *norm = -n1;
200  /*
201  * 64 bit normalization
202  */
203  *normalized = (((uint32_t) lo) >> n1) | (hi << (32 - n1));
204  }
205 }

207 __STATIC_INLINE q31_t arm_div_q63_to_q31(q63_t num, q31_t den)
208 {
209  q31_t result;
210  uint64_t absNum;
211  int32_t normalized;
212  int32_t norm;
213 
214  /*
215  * if sum fits in 32bits
216  * avoid costly 64-bit division
217  */
218  absNum = num > 0 ? num : -num;
219  arm_norm_64_to_32u(absNum, &normalized, &norm);
220  if (norm > 0)
221  /*
222  * 32-bit division
223  */
224  result = (q31_t) num / den;
225  else
226  /*
227  * 64-bit division
228  */
229  result = (q31_t) (num / den);
230 
231  return result;
232 }
