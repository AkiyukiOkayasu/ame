196 __STATIC_FORCEINLINE arm_status arm_sqrt_f32(
197  float32_t in,
198  float32_t * pOut)
199  {
200  if (in >= 0.0f)
201  {
202 #if defined ( __CC_ARM )
203  #if defined __TARGET_FPU_VFP
204  *pOut = __sqrtf(in);
205  #else
206  *pOut = sqrtf(in);
207  #endif
208 
209 #elif defined ( __ICCARM__ )
210  #if defined __ARMVFP__
211  __ASM("VSQRT.F32 %0,%1" : "=t"(*pOut) : "t"(in));
212  #else
213  *pOut = sqrtf(in);
214  #endif
215 
216 #else
217  *pOut = sqrtf(in);
218 #endif
219 
220  return (ARM_MATH_SUCCESS);
221  }
222  else
223  {
224  *pOut = 0.0f;
225  return (ARM_MATH_ARGUMENT_ERROR);
226  }
227  }
