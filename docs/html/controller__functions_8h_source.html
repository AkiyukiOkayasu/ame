268  __STATIC_FORCEINLINE float32_t arm_pid_f32(
269  arm_pid_instance_f32 * S,
270  float32_t in)
271  {
272  float32_t out;
273 
274  /* y[n] = y[n-1] + A0 * x[n] + A1 * x[n-1] + A2 * x[n-2] */
275  out = (S->A0 * in) +
276  (S->A1 * S->state[0]) + (S->A2 * S->state[1]) + (S->state[2]);
277 
278  /* Update state */
279  S->state[1] = S->state[0];
280  S->state[0] = in;
281  S->state[2] = out;
282 
283  /* return to application */
284  return (out);
285 
286  }

301 __STATIC_FORCEINLINE q31_t arm_pid_q31(
302  arm_pid_instance_q31 * S,
303  q31_t in)
304  {
305  q63_t acc;
306  q31_t out;
307 
308  /* acc = A0 * x[n] */
309  acc = (q63_t) S->A0 * in;
310 
311  /* acc += A1 * x[n-1] */
312  acc += (q63_t) S->A1 * S->state[0];
313 
314  /* acc += A2 * x[n-2] */
315  acc += (q63_t) S->A2 * S->state[1];
316 
317  /* convert output to 1.31 format to add y[n-1] */
318  out = (q31_t) (acc >> 31U);
319 
320  /* out += y[n-1] */
321  out += S->state[2];
322 
323  /* Update state */
324  S->state[1] = S->state[0];
325  S->state[0] = in;
326  S->state[2] = out;
327 
328  /* return to application */
329  return (out);
330  }

347 __STATIC_FORCEINLINE q15_t arm_pid_q15(
348  arm_pid_instance_q15 * S,
349  q15_t in)
350  {
351  q63_t acc;
352  q15_t out;
353 
354 #if defined (ARM_MATH_DSP)
355  /* Implementation of PID controller */
356 
357  /* acc = A0 * x[n] */
358  acc = (q31_t) __SMUAD((uint32_t)S->A0, (uint32_t)in);
359 
360  /* acc += A1 * x[n-1] + A2 * x[n-2] */
361  acc = (q63_t)__SMLALD((uint32_t)S->A1, (uint32_t)read_q15x2 (S->state), (uint64_t)acc);
362 #else
363  /* acc = A0 * x[n] */
364  acc = ((q31_t) S->A0) * in;
365 
366  /* acc += A1 * x[n-1] + A2 * x[n-2] */
367  acc += (q31_t) S->A1 * S->state[0];
368  acc += (q31_t) S->A2 * S->state[1];
369 #endif
370 
371  /* acc += y[n-1] */
372  acc += (q31_t) S->state[2] << 15;
373 
374  /* saturate the output */
375  out = (q15_t) (__SSAT((q31_t)(acc >> 15), 16));
376 
377  /* Update state */
378  S->state[1] = S->state[0];
379  S->state[0] = in;
380  S->state[2] = out;
381 
382  /* return to application */
383  return (out);
384  }

467 __STATIC_FORCEINLINE void arm_park_q31(
468  q31_t Ialpha,
469  q31_t Ibeta,
470  q31_t * pId,
471  q31_t * pIq,
472  q31_t sinVal,
473  q31_t cosVal)
474  {
475  q31_t product1, product2; /* Temporary variables used to store intermediate results */
476  q31_t product3, product4; /* Temporary variables used to store intermediate results */
477 
478  /* Intermediate product is calculated by (Ialpha * cosVal) */
479  product1 = (q31_t) (((q63_t) (Ialpha) * (cosVal)) >> 31);
480 
481  /* Intermediate product is calculated by (Ibeta * sinVal) */
482  product2 = (q31_t) (((q63_t) (Ibeta) * (sinVal)) >> 31);
483 
484 
485  /* Intermediate product is calculated by (Ialpha * sinVal) */
486  product3 = (q31_t) (((q63_t) (Ialpha) * (sinVal)) >> 31);
487 
488  /* Intermediate product is calculated by (Ibeta * cosVal) */
489  product4 = (q31_t) (((q63_t) (Ibeta) * (cosVal)) >> 31);
490 
491  /* Calculate pId by adding the two intermediate products 1 and 2 */
492  *pId = __QADD(product1, product2);
493 
494  /* Calculate pIq by subtracting the two intermediate products 3 from 4 */
495  *pIq = __QSUB(product4, product3);
496  }
