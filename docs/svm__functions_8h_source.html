64 __STATIC_INLINE float32_t arm_exponent_f32(float32_t x, int32_t nb)
65 {
66  float32_t r = x;
67  nb --;
68  while(nb > 0)
69  {
70  r = r * x;
71  nb--;
72  }
73  return(r);
74 }
