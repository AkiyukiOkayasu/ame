135 template <class T>
136 class AnyPattern {
137  protected:
138  // Pattern data
139  T *data;
140  // To know if the pattern has loaded any data
141  bool isLoaded;
142  // Memory generation when the data was loaded.
143  // If memory generation is different when accessing the data
144  // the pattern should return NULL.
145  unsigned long currentGen;
146  PatternMgr *m_mgr;
147  // Nb of samples in the loaded pattern
148  Testing::nbSamples_t m_nbSamples;
149  public:
150  AnyPattern()
151  {
152  this->data = NULL;
153  this->isLoaded = false;
154  this->currentGen = 0;
155  this->m_mgr=NULL;
156  this->m_nbSamples = 0;
157  }
158 
159  bool isTailEmpty()
160  {
161  if (m_mgr)
162  {
163  return(m_mgr->IsTailEmpty((char*)this->ptr(),this->nbSamples()*sizeof(T)));
164  }
165  else
166  {
167  return(true);
168  }
169  }
170 
179  T *ptr()
180  {
181  if (this->m_mgr == NULL)
182  {
183  return(NULL);
184  }
185 
186  if (this->currentGen != this->m_mgr->generation())
187  {
188  return(NULL);
189  }
190  else
191  {
192  if (this->isLoaded)
193  {
194  if (this->m_nbSamples > 0)
195  {
196  return(this->data);
197  }
198  else
199  {
200  return(NULL);
201  }
202  }
203  else
204  {
205  return(NULL);
206  }
207  }
208  }
209 
210  Testing::nbSamples_t nbSamples()
211  {
212  if (this->m_mgr == NULL)
213  {
214  return(0);
215  }
216 
217  if (this->currentGen != this->m_mgr->generation())
218  {
219  return(0);
220  }
221  if (this->isLoaded)
222  {
223  return(this->m_nbSamples);
224  }
225  else
226  {
227  return(0);
228  }
229  }
230 
231 };
