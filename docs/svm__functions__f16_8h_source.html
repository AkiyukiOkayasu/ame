66 __STATIC_INLINE float16_t arm_exponent_f16(float16_t x, int32_t nb)
67 {
68  float16_t r = x;
69  nb --;
70  while(nb > 0)
71  {
72  r = r * x;
73  nb--;
74  }
75  return(r);
76 }
