84 template <typename FloatType>
85 void process (AudioBlockView<FloatType>& block)
86 {
87 if (playing.load() == false)
88 {
89 return;
90 }
91
92 assert (block.getNumChannels() >= numChannels);
93 assert (0 <= readPosition.load() && readPosition <= numSamples);
94
96 switch (formatTag)
97 {
98 case fmt::wFormatTag::PCM:
99 {
100 auto offset = (bitRate / 8) * numChannels * readPosition.load();
101 for (uint_fast32_t samp = 0; samp < block.getNumSamples(); ++samp)
102 {
103 if (readPosition + samp >= numSamples)
104 {
106 readPosition.store (0);
107 playing.store (false);
108 return;
109 }
110
111 for (uint_fast32_t ch = 0; ch < block.getNumChannels(); ++ch)
112 {
113 int32_t temp = 0;
114 for (uint32_t b = 32 - bitRate; b < 32; b += 8) //bitDepthが16bitや24bitのときは下部を0埋めする.
115 {
116 temp |= (dataChunk.data[offset] << b);
117 ++offset;
118 }
119 const FloatType s = static_cast<FloatType> (temp) / normalizeFactor;
120 block.addSample (ch, samp, s);
121 }
122 }
123 readPosition += block.getNumSamples();
124 break;
125 }
126 case fmt::wFormatTag::ImaAdpcm:
127 assert (false);
128 break;
129 case fmt::wFormatTag::IeeeFloat:
130 assert (false);
131 break;
132 default:
133 assert (false); //invalid wFormat
134 break;
135 }
