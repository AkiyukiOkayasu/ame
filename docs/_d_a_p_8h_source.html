287 #if defined(__CC_ARM)
288 __STATIC_FORCEINLINE void PIN_DELAY_SLOW (uint32_t delay) {
289  uint32_t count = delay;
290  while (--count);
291 }
292 #else
293 __STATIC_FORCEINLINE void PIN_DELAY_SLOW (uint32_t delay) {
294  __ASM volatile (
295  ".syntax unified\n"
296  "0:\n\t"
297  "subs %0,%0,#1\n\t"
298  "bne 0b\n"
299  : "+l" (delay) : : "cc"
300  );
301 }
